DROP DATABASE ALUGUEL_IMOVEIS;
CREATE DATABASE ALUGUEL_IMOVEIS;
USE ALUGUEL_IMOVEIS;

CREATE TABLE Imovel (
Imovel_id INTEGER PRIMARY KEY AUTO_INCREMENT,
Tipo VARCHAR(12),
Cidade VARCHAR(25),
Bairro VARCHAR(25),
CEP VARCHAR(9),
Rua VARCHAR(50),
Numero INTEGER,
Imovel_situacao VARCHAR(10),
Qtd_quarto INTEGER,
Tamanho DECIMAL(10),
Observacao VARCHAR(160),
Valor_locacao DECIMAL(10)
);

CREATE TABLE Cliente (
Cliente_id INTEGER PRIMARY KEY AUTO_INCREMENT,
cliente_nome VARCHAR(40),
Data_nascimento DATE,
CPF varchar(11),
sexo CHAR(1)
);


CREATE TABLE Locacao (
locacao_id INTEGER PRIMARY KEY AUTO_INCREMENT,
Data_fim_contrato DATE,
Observacao VARCHAR(200),
Data_ini_contrato DATE,
Locacao_situacao VARCHAR(10),
Valor_reajustado DECIMAL(10),
Cliente_id INTEGER,
Imovel_id INTEGER,
FOREIGN KEY(Cliente_id) REFERENCES Cliente (Cliente_id),
FOREIGN KEY(Imovel_id) REFERENCES Imovel (Imovel_id)
);

CREATE TABLE Debito_automatico (
Debito_id INTEGER PRIMARY KEY AUTO_INCREMENT,
Banco VARCHAR(10),
Conta INTEGER,
Agencia INTEGER,
Operacao INTEGER,
Debito_situacao VARCHAR(10),
Data_debito DATE,
locacao_id INTEGER,
Cliente_id INTEGER,
FOREIGN KEY(locacao_id) REFERENCES Locacao (locacao_id),
FOREIGN KEY(Cliente_id) REFERENCES Cliente (Cliente_id)
);
